月份(单词简写)转数字

Written by Hiroshi.tao with ♥ on September 27, 2017 in 开发 52 words
function Monthly2Number() {
    case "$1" in
        Jan) echo 1;;
        Feb) echo 2;;
        Mar) echo 3;;
        Apr) echo 4;;
        May) echo 5;;
        Jun) echo 6;;
        Jul) echo 7;;
        Aug) echo 8;;
        Sep) echo 9;;
        Oct) echo 10;;
        Nev) echo 11;;
        Dec) echo 12;;
        *)   exit 2;;
    esac
}
